class Solution:
    def isPowerOfThree(self, n: int) -> bool:

        #let n=3**t

        if n<=0:              # log function definition doesn't allow n<=0
            return False
        else:
            t= log(n,3)       #log of n to the base 3
            if t-int(t)==0:   #checking if fraction point exists, i.e. if t is and int or not
                return True
            else:
                return False
