class Solution:
    def containsDuplicate(self, nums: List[int]) -> bool:
 
        nums.sort()
        n=len(nums)

        c=0
        for i in range(1,n):
            if nums[i-1]==nums[i]:
                return True
                break

            else:
                c=c+1
            
        
        if c==(n-1):
            return False

 class Solution:
    def containsDuplicate(self, nums: List[int]) -> bool:
        

        n=len(nums)    
        arr=[]

        nums.sort()
        for i in range(n):
            l=nums[i]
            arr.append(l)

        c=0
        for i in range(1,n):
            if arr[i-1]==arr[i]:
                nums.pop(i-c)
                c=c+1

        k=len(nums)
        m=len(arr)

        return m!=k
