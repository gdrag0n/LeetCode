# Approach 01
```
class Solution:
    def findDuplicate(self, nums: List[int]) -> int:

        for i in nums:
            if nums.count(i)>1:
                return i
```

```
class Solution:
    def findDuplicate(self, nums: List[int]) -> int:

        n=len(nums)
        return sum(nums)-sum(list(range(n)))
```
