class Solution:
    def reverseVowels(self, s: str) -> str:

        s=list(s)
        v='aeiou'                                     #required to handle case sensitivity
        l=[]

        for i in range(len(s)):
            if (s[i] in v) or (s[i] in v.upper()):    #upper() does not work on list, works on string
                l.append(i)
            
        n=len(l)
        for i in range(n//2):
            a=l[i]
            b=l[n-i-1]
            s[a], s[b] = s[b], s[a]
        
        ans=''
        for i in s:
            ans=ans+i

        return ans
