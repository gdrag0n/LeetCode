class Solution:
    def maxProfit(self, prices: List[int]) -> int:

        l=[]
        n=len(prices)
        for i in range(n):
            for j in range(i+1, n):
                ans= prices[j]-prices[i]
                l.append(ans)

        if n>1:
            if max(l)<=0:
                return 0
            else:
                return max(l)
        elif n>10**5:
            return 0
        else:
            return 0
