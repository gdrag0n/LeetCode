class Solution:

    def runningSum(self, nums: List[int]) -> List[int]:

        rs=[]
        c=0

        for i in range(len(nums)):
            c=c+nums[i]
            rs.append(c)

        return rs

class Solution:
    def runningSum(self, nums: List[int]) -> List[int]:

        rs=[]
        rs.append(nums[0])

        for i in range(1,len(nums)):
            rs.append(nums[i]+rs[i-1])
        
        return rs
